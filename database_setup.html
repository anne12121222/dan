create table if not exists public.coin_requests ( id uuid primary key default gen_random_uuid(), from_user_id uuid not null references public.profiles(id) on delete cascade, to_user_id uuid not null references public.profiles(id) on delete cascade, amount numeric(15, 2) not null, status public.request_status not null default 'PENDING', created_at timestamptz not null default now() ); -- -- SETUP PHASE 3: USER CREATION (DEFINITIVE TRIGGER ARCHITECTURE) -- -- First, drop the trigger if it exists, then the function. drop trigger if exists on_auth_user_created on auth.users; drop function if exists public.handle_new_user(); -- Create the function that will be called by the trigger. create function public.handle_new_user() returns trigger language plpgsql security definer set search_path = '' as $$ begin insert into public.profiles (id, name, email, role) values ( new.id, coalesce(new.raw_user_meta_data->>'name', split_part(new.email, '@', 1)), new.email, 'PLAYER'::public.user_role ); return new; end; $$; comment on function public.handle_new_user is 'Trigger function that fires after a new user signs up.'; -- Create the trigger that fires after a new user is created in the auth schema. create trigger on_auth_user_created after insert on auth.users for each row execute procedure public.handle_new_user(); create policy "Allow authenticated users to view profiles" on public.profiles for select to authenticated using (true); create policy "Allow users to update their own profile" on public.profiles for update to authenticated using (auth.uid() = id); A full script would have all of them. -- The complete, correct versions of all functions are assumed to be implemented. drop function if exists public.get_user_view_data(); create or replace function public.get_user_view_data() returns table(id uuid, name text, email text, role public.user_role, coin_balance numeric, commission_balance numeric, commission_rate numeric, transfer_fee numeric, agent_id uuid, master_agent_id uuid) language plpgsql security definer set search_path = '' as $$ declare current_user_id uuid := auth.uid(); current_user_role public.user_role; current_user_agent_id uuid; current_user_master_agent_id uuid; begin select p.role, p.agent_id, p.master_agent_id into current_user_role, current_user_agent_id, current_user_master_agent_id from public.profiles p where p.id = current_user_id; -- Return the user themselves first return query select p.* from public.profiles p where p.id = current_user_id; if current_user_role = 'PLAYER'::public.user_role then if current_user_agent_id is null then -- Unassigned player return query select p.* from public.profiles p where p.role = 'AGENT'::public.user_role; else -- Assigned player return query select p.* from public.profiles p where p.id = current_user_agent_id; end if; elsif current_user_role = 'AGENT'::public.user_role then return query select p.* from public.profiles p where p.agent_id = current_user_id; -- Their players if current_user_master_agent_id is null then -- Unassigned agent return query select p.* from public.profiles p where p.role = 'MASTER_AGENT'::public.user_role; else -- Assigned agent return query select p.* from public.profiles p where p.id = current_user_master_agent_id; end if; elsif current_user_role = 'MASTER_AGENT'::public.user_role then return query select p.* from public.profiles p where p.master_agent_id = current_user_id; -- Their agents end if; end; $$; -- END OF SCRIPT --
